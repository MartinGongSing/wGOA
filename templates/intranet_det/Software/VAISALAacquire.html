#/bin/bash
#VAISALAacquire
#11 Fev 2017
stty -F /dev/ttyUSB0 19200 -cstopb cs8 -parenb -crtscts -ixon -ixoff onlcr
# -F open and use the specified DEVICE instead of stdin
# -cstopb use 1 stop bit per character
# cs8 set character size to 8 bits
# -parenb generate parity bit in output and expect parity bit in input
# -crtscts enable RTS/CTS handshaking
# -ixon enable XON/XOFF flow control
# -ixoff enable sending of start/stop characters
# onlcr translate newline to carriage return-newline
echo "Power-up the Vaisala HMT330 [ENTER]:"
read
echo -e "S\n\r" > /dev/ttyUSB0
# -e enable interpretation of backslash escapes
echo -e "DATE" > /dev/ttyUSB0
echo "Enter new date [yyyy-mm-dd]:"
read "data"
echo -e "$data" > /dev/ttyUSB0
echo -e "TIME" > /dev/ttyUSB0
echo "Enter time [hh:mm:ss]:"
read "time"
echo -e "$time" > /dev/ttyUSB0
echo "Enter file name [*.txt]:"
read "filename"
echo -e "R\n\r" > /dev/ttyUSB0
while true
do
read X < /dev/ttyUSB0
echo $X
p="$X\n\r"
echo $X >> $filename
echo $X > Tape_Control_File.out
done
